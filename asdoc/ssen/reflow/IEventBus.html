eventBusSample.txt
[Inject]
public var eventBus:IEventBus;

private var dataUpdate:IEventListener;
private var dataCreate:IEventListener;

// Automatically execute after dependency injection
[PostConstruct]
public function injectionComplete():void {
    // add event listener
    dataUpdate = eventBus.on(DataEvent.UPDATE, dataUpdateHandler);  
    dataCreate = eventBus.on(DataEvent.CREATE, dataCreateHandler);
}

public function dispose():void {
    // remove event listener
    dataUpdate.remove();
    dataCreate.remove();
    
    dataUpdate = null;
    dataCreate = null;
}

// event handlers
private function dataUpdateHandler(event:DataEvent):void {
}

private function dataCreateHandler(event:DataEvent):void {
}
